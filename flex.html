Grow

1
2
3
					
	<div class="flex">
		<div>
			1 <!-- This item will not grow -->
		</div>
		<div class="flex-grow">
			2 <!-- This item will grow -->
		</div>
		<div>
			3 <!-- This item will not grow -->
		</div>
	</div>

Don't grow

		</div>
